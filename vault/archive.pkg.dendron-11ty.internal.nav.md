- shortcode.js
```js
toMarkdown {
	navParent = getClosetNavVisibleParent
}


getClosetNavVisibleParent(fname, noteIndex) {
	maybeNote := getByFname(fname)
	if !maybeNote
		return noteIndex

	if maybeNote.permalink = '/'
		return 'root'

	if notes[maybeNote.parent].has_collection
		return notes[maybeNote.parent]
	else
		return maybeNote
}
```
